import urllib2
import re
from bs4 import BeautifulSoup
import lxml
import  json


def finaltask (url):
    if url =="http://www.sxrtv.com//content/d/f/2012-11-1/1352793900795.shtml":
     return

    response = urllib2.urlopen(url)
    html = response.read()
    soup = BeautifulSoup(html,"lxml")
    x = soup.find_all('script')[9]
    json_url = r"http://www.sxrtv.com"+str(x)[50:-168].replace( "',channel:\"3346",'')

    # print json_url
    response = urllib2.urlopen(json_url)
    # print response.read()
    # data = json.loads(response.read())
    data = json.load(response)
    # data = json.loads(r.read().decode(r.info().get_param('charset') or 'utf-8'))
    # print data
    # print data.keys()
    print data['videos'][0]['url']

    return

def printme( iterator ):
    # print iterator
    response = urllib2.urlopen(iterator)
    html = response.read()
    regex = "(?<=href=\").+?(?=\")|(?<=href=\').+?(?=\')"
    pattern = re.compile(r'/content/d/f/20*')
    link_list =re.finditer(regex ,html)
    for url in link_list:
        match = pattern.match(url.group())
        if match:
            finaltask('http://www.sxrtv.com/'+url.group())

            return
            #     print  "http://www.sxrtv.com/"+url.group()
    # print "All Tasks Finished! "
    return
